(function() {
	
	'use strict';
	var highlightAuthor = 'regattanews.com';
	
	Livefyre.require([
	    'streamhub-wall#3.4.0',
	    'streamhub-sdk#2.16.0'
	], function(LiveMediaWall, SDK) { 
	    var oldCreateContentView,
			contentListView,
			collection,
			wall;
		collection = new (SDK.Collection)({
	        network: 'cnn.fyre.co',
	        siteId: 374065,
	        articleId: 'custom-1439230213111',
	        environment: 'livefyre.com'
	    });
	    wall = new LiveMediaWall({
	        el: document.getElementById('lf-media-wall'),
	        initial: 12,
	        bodyFontSize: '16px',
	        bodyLineHeight: '22px',
	      	buttonTextColor: '#4D8FE0',
	      	cardBackgroundColor: '#404040',
	      	displayNameColor: '#4D8FE0',
	      	fontFamily: 'Helvetica',
	      	footerTextColor: '#fff',
	      	linkAttachmentTextColor: '#4D8FE0',
	      	linkColor: '#4D8FE0',
	        textColor: '#fff',
	        usernameColor: '#fff'
	    });
	    contentListView = wall._wallView;
	    oldCreateContentView = contentListView.createContentView;
	    contentListView.createContentView = function() {
	        var contentView = oldCreateContentView.apply(this, arguments),
	        	oldRender = contentView.render;
	        contentView.render = function() {
	            this.$el.find('.content-author-name :contains(' + highlightAuthor + ')').closest('.content').addClass('content-sponsored');
	            return oldRender.apply(this, arguments);
	        };
	        
	        return contentView;
	    };
	    collection.pipe(wall);
	});
})();
