		var a_temp_cookie = '';
		var ck_name = '';
		var cookie_value = '';
		var b_cookie_found = false;

			a_temp_cookie = a_all_cookies[i].split( '=' );
			ck_name = a_temp_cookie[0].replace(/^\s+|\s+$/g, '');
			if ( ck_name == check_name ) {
				b_cookie_found = true;
				if ( a_temp_cookie.length > 1 )	{
					cookie_value = unescape( a_temp_cookie[1].replace(/^\s+|\s+$/g, '') );
				}
				return cookie_value;
				break;
			}
			a_temp_cookie = null;
			ck_name = '';
		}
		if ( !b_cookie_found ) {
			return null;
		}
	}
